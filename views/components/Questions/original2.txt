    <script>
        const addQuestionBtn = document.getElementById('add-question');
        const generateQuizBtn = document.getElementById('generate-quiz');
        const quizContainer = document.getElementById('quiz-container');
        let questionCounter = `<%= data1.questions.length %>`;

        addQuestionBtn.addEventListener('click', () => {
            questionCounter++;
            const newQuestion = document.createElement('div');
            newQuestion.classList.add('question');
            newQuestion.innerHTML = `
                <label for="question${questionCounter}">Question ${questionCounter}:</label>
                <textarea id="question${questionCounter}" class="questionText" name="question${questionCounter}" placeholder="How you doing?"></textarea>
                <div class="options">
                    <div class="option">
                        <input type="text" class="optiontext" name="option${questionCounter}-1" placeholder="Option 1">
                        <button class="remove-option" disabled>-</button>
                    </div>
                    <div class="option">
                        <input type="text" class="optiontext" name="option${questionCounter}-2" placeholder="Option 2">
                        <button class="remove-option" disabled>-</button>
                    </div>
                    <button class="add-option">Add option</button>
                </div>
            `;
            quizContainer.appendChild(newQuestion);
        });

        quizContainer.addEventListener('click', (event) => {
            if (event.target.classList.contains('add-option')) {
                const optionsContainer = event.target.parentNode;
                const optionCounter = optionsContainer.querySelectorAll('.option').length + 1;
                const newOption = document.createElement('div');
                newOption.classList.add('option');
                newOption.innerHTML = `
                    <input type="text" class="optiontext" name="option${questionCounter}-${optionCounter}" placeholder="Option ${optionCounter}">
                    <button class="remove-option">-</button>
                `;
                optionsContainer.insertBefore(newOption, event.target);
                checkOptionButtons(optionsContainer);
            } else if (event.target.classList.contains('remove-option')) {
                const optionsContainer = event.target.parentNode.parentNode;
                event.target.parentNode.remove();
                checkOptionButtons(optionsContainer);
            }
        });

        function checkOptionButtons(optionsContainer) {
            const optionButtons = optionsContainer.querySelectorAll('.remove-option');
            optionButtons.forEach(button => {
                if (optionButtons.length > 2) {
                    button.removeAttribute('disabled');
                } else {
                    button.setAttribute('disabled', 'disabled');
                }
            });
        }

        generateQuizBtn.addEventListener('click', () => {
            const id = <%- JSON.stringify(data1.set_id) %>;
            const owner_id = <%- JSON.stringify(data1.owner_id) %>;
            const quizData = [];
            const questions = quizContainer.querySelectorAll('.question');
            questions.forEach((questionElement, index) => {
                const questionText = questionElement.querySelector(`textarea[id="question${index + 1}"]`).value;
                const options = [];
                const optionElements = questionElement.querySelectorAll('.option input[type="text"]');
                optionElements.forEach(optionElement => {
                    options.push(optionElement.value);
                });
                quizData.push({
                    question: questionText,
                    options: options,
                    set_id: id
                });
            });
            console.log(quizData);
            const setname = document.getElementById('setname');
            
            fetch(`/question?set_id=${id}&owner_id=${owner_id}&setname=${setname.value}`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify(quizData)
            })
            .then(quizData => {
                console.log('Success:', quizData);
            })
            .catch((error) => {
                console.error('Error:', error);
            });
        });

        function sendData() {
            fetch('/submit-data', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify(data)
            })
            .then(response => response.json())
            .then(data => {
                console.log('Success:', data);
            })
            .catch((error) => {
                console.error('Error:', error);
            });
        }
    </script>
